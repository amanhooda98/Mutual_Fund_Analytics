def beta_calc(start_date, end_date, type_of_mf):
    index_start_value = result_df.loc[
        result_df["date"] == start_date, "index_nrml"
    ].values[0]
    index_end_value = result_df.loc[result_df["date"] == end_date, "index_nrml"].values[
        0
    ]
    index_change = index_start_value - index_end_value

    if type_of_mf == "equity":
        mf_start_value = result_df.loc[
            result_df["date"] == start_date, "avg_equity"
        ].values[0]
        mf_end_value = result_df.loc[
            result_df["date"] == end_date, "avg_equity"
        ].values[0]
        result = (mf_start_value - mf_end_value) / index_change

    elif type_of_mf == "debt":
        mf_start_value = result_df.loc[
            result_df["date"] == start_date, "avg_debt"
        ].values[0]
        mf_end_value = result_df.loc[result_df["date"] == end_date, "avg_debt"].values[
            0
        ]
        result = (mf_start_value - mf_end_value) / index_change

    elif type_of_mf == "hybrid":
        mf_start_value = result_df.loc[
            result_df["date"] == start_date, "avg_hybrid"
        ].values[0]
        mf_end_value = result_df.loc[
            result_df["date"] == end_date, "avg_hybrid"
        ].values[0]
        result = (mf_start_value - mf_end_value) / index_change
    else:
        result = "Wrong mf Type"

    return round(result, 2)
